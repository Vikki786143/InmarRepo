1. SELECT * FROM Customer Where Name like 'Joe%';

2. select P.Name FROM Customer C INNER JOIN Order O 
ON O.CustomerID = C.CustomerID
JOIN OrderProduct OP ON OP.OrderID = O.OrderID
JOIN  Prodcut P ON P.ProdcutID = OP.ProdcutID
where C.Name = 'Joe' and CAST(O.CreatedAT as date) = '11/1/2016'

3. select P.Name,SUM(P.Price) FROM Customer C INNER JOIN Order O 
ON O.CustomerID = C.CustomerID
JOIN OrderProduct OP ON OP.OrderID = O.OrderID
JOIN  Prodcut P ON P.ProdcutID = OP.ProdcutID
where C.Name = 'Joe' 


4.select C.Name FROM Customer C INNER JOIN Order O 
ON O.CustomerID = C.CustomerID
JOIN OrderProduct OP ON OP.OrderID = O.OrderID
JOIN  Prodcut P ON P.ProdcutID = OP.ProdcutID
GROUP BY C.Name,O.OrderID
HAving count(O.OrderID) > 1
